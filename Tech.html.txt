    <script>

        const produtos = [
            {id:1,nome:"Notebook Gamer",preco:3500},
            {id:2,nome:"Mouse Gamer",preco:150},
            {id:3,nome:"Teclado Mecânico",preco:250},
            {id:4,nome:"Monitor 24''",preco:900},
            {id:5,nome:"Headset",preco:200},
            {id:6,nome:"Webcam",preco:180},
            {id:7,nome:"Microfone USB",preco:300},
            {id:8,nome:"SSD 1TB",preco:450},
            {id:9,nome:"HD Externo",preco:400},
            {id:10,nome:"Impressora",preco:600},
            {id:11,nome:"Roteador Wi-Fi",preco:250},
            {id:12,nome:"Placa de Vídeo",preco:2500},
            {id:13,nome:"Processador",preco:1200},
            {id:14,nome:"Memória RAM",preco:300},
            {id:15,nome:"Fonte 600W",preco:350},
            {id:16,nome:"Gabinete Gamer",preco:500},
            {id:17,nome:"Notebook Básico",preco:2200},
            {id:18,nome:"Tablet",preco:1300},
            {id:19,nome:"Smartwatch",preco:700},
            {id:20,nome:"Caixa de Som",preco:280}
        ];

        let carrinho = [];

        const areaProdutos = document.getElementById("produtos");

        produtos.forEach(produto => {

            areaProdutos.innerHTML += `
                <div class="card">
                    <h3>${produto.nome}</h3>
                    <p>R$ ${produto.preco}</p>
                    <button class="adicionar"
                    onclick="adicionarItem(${produto.id})">
                    Adicionar
                    </button>
                </div>
            `;
        });

        function adicionarItem(id){

            let produto = produtos.find(p => p.id === id);

            let item = carrinho.find(i => i.id === id);

            if(item){
                item.quantidade++;
            }else{
                carrinho.push({
                    ...produto,
                    quantidade:1
                });
            }

            atualizarCarrinho();
        }

        function removerItem(id){

            carrinho = carrinho.filter(item => item.id !== id);

            atualizarCarrinho();
        }

        function limparCarrinho(){

            carrinho = [];

            atualizarCarrinho();
        }

        function atualizarCarrinho(){

            let lista = document.getElementById("listaCarrinho");

            let total = 0;
            let quantidadeTotal = 0;

            lista.innerHTML = "";

            carrinho.forEach(item => {

                total += item.preco * item.quantidade;
                quantidadeTotal += item.quantidade;

                lista.innerHTML += `
                    <div class="item">
                        <strong>${item.nome}</strong><br>
                        Quantidade: ${item.quantidade}<br>
                        Total: R$ ${item.preco * item.quantidade}
                        <br>
                        <button class="remover"
                        onclick="removerItem(${item.id})">
                        Remover
                        </button>
                    </div>
                `;
            });

            document.getElementById("total").innerHTML =
                "Total: R$ " + total;

            document.getElementById("contador").innerHTML =
                quantidadeTotal;
        }
